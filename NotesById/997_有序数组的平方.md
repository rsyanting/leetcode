public int[] sortedSquares(int[] A) {
    int len = A.length;
    int[] res = new int[len];
    int left = 0, right = len - 1;
    int idx = len - 1;
    while (left <= right) {
        if (Math.abs(A[left]) <= Math.abs(A[right])) {
            res[idx] = (int) Math.pow(Math.abs(A[right]), 2);
            right--;
        } else {
            res[idx] = (int) Math.pow(Math.abs(A[left]), 2);
            left++;
        }
        idx--;
    }

    return res;
}
