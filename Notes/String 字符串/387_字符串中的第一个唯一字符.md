public int firstUniqChar(String s) {
    int[] keys = new int[26];
    for (char key : s.toCharArray()) {
        keys[key - 'a']++;
    }
    for (int i = 0; i < s.length(); i++) {
        if (keys[s.charAt(i) - 'a'] == 1) {
            return i;
        }
    }
    return -1;
}
