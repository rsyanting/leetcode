public boolean isValidBST(TreeNode root) {
    Stack<TreeNode> stack = new Stack<>();
    long inorderValue = Long.MIN_VALUE;

    while (!stack.isEmpty() || root != null) {
        while (root != null) {
            stack.push(root);
            root = root.left;
        }
        root = stack.pop();
        if (root.val <= inorderValue) {
            return false;
        }
        inorderValue = root.val;
        root = root.right;
    }
    return true;
}
